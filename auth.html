document.addEventListener('DOMContentLoaded', () => { const loginBtn = document.getElementById('login-btn'); const registerBtn = document.getElementById('register-btn'); const message = document.getElementById('message'); async function handleAuth(type) { const email = document.getElementById('email').value; const password = document.getElementById('password').value; if (!email || !password) return message.innerText = "Bilgileri girin!"; message.innerText = "İşleniyor..."; // Supabase işlemini başlat const { data, error } = type === 'login' ? await window.supabase.auth.signInWithPassword({ email, password }) : await window.supabase.auth.signUp({ email, password }); // Hata varsa konsola ve ekrana bas if (error) { console.error("Supabase Hatası:", error); message.innerText = "Hata: " + error.message; } else { console.log("Supabase Yanıtı:", data); // Başarılıysa veriyi gör if (type === 'login') { message.innerText = "Giriş başarılı!"; setTimeout(() => window.location.href = 'index.html', 1000); } else { message.innerText = "Kayıt başarılı! Şimdi giriş yapabilirsiniz."; } } } if(loginBtn) loginBtn.onclick = () => handleAuth('login'); if(registerBtn) registerBtn.onclick = () => handleAuth('register'); });
